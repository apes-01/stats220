My meme

Inspired by the 2001 Legally Blonde star actress, Reese Witherspoon and a pre-existing meme. My university mood speaks for itself below.

library(magick)

# reading in our 2 images for our meme - scale our images to be similar size

legallyblond <- image_read("https://elle-backend.imgix.net/media/article/legally-blonde-life-lessions-elle-2018.jpg?ixlib=vue-2.9.1&auto=format&ar=414%3A414&fit=crop&fp-x=0.5&fp-y=0.5&w=1678") %>%
  image_scale(200)

notlegallyblond <- image_read("https://i.dailymail.co.uk/i/pix/2011/03/15/article-0-0B2E136A00000578-572_468x647.jpg") %>%
  image_scale(200) 


# we then, append our two images above together

image <- c(legallyblond, notlegallyblond) 
images <- image_append(image, stack = T)


# lets make our two different text for our meme

text1 <- image_blank(width = 200, height = 200,
                    color = "#FFFFFFFF") %>%
  image_annotate(text = "me on the first \n day of uni",
                 color="#000000",
                 size = 20,
                 gravity = "center")

text2 <- image_blank(width = 200, height = 276,
                     color = "#FFFFFFFF") %>%
  image_annotate(text = " vs. me on the last \n semester",
                 color="#000000",
                 size = 20,
                 gravity = "center")


# now lets append our texts together

textz <- c(text1,text2)

textes <- image_append(textz, stack = T)


# lets put our meme together - our images and textes to be side-by-side

mymeme <- c(images, textes) %>%
  image_append() %>% 
  image_scale(400) %>%
  image_write("mymeme.png")
